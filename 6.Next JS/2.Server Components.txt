import notfound from "@/app/not-found";
import React from "react";
interface IDType {
  params: Promise<{ id: number }>;
  searchParams: Promise<{}>;
}
export default async function page(props: IDType) {
  const { id } = await props.params;
  if (id >= 1000) {
    return notfound();
  }
  return (
    <div>
      <h1>id: {id}</h1>
    </div>
  );
}
