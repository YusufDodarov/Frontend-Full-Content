import { useState, useMemo } from "react";

export default function App() {
  const heavy = useMemo(() => {
    let sum = 0;
    for (let i = 0; i <= 1000000000; i++) {
      sum += i;
    }
    return sum;
  }, []);
  const [count, setCount] = useState(0);
  console.log("render");
  return (
    <div>
      <p>{heavy}</p>
      <p>{count}</p>
      <button onClick={()=>setCount(count+1)}>+</button>
    </div>
  );
}
