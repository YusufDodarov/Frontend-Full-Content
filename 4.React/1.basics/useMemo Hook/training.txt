import { useState } from "react";
export default function App() {
  const [count, setCount] = useState(0);
  const heavy = () => {
    let sum=0;
    for (let i = 0; i <= 1000000000; i++) {
      sum+=i
    }
    return sum
  };
  let result = heavy()
  console.log("render");
  return (
    <div>
      <p>{result}</p>
      <p>{count}</p>
      <button onClick={()=>setCount(count + 1)}>+</button>
    </div>
  );
}
